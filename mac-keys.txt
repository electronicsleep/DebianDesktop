# Change Linux keys to Mac like keybindings (for mac users)
# Having the same keys as my 2015 MacBook Pro is convenient

apt-get install gnome-tweaks -y

# Ubuntu: Tweaks -> Keyboard & Mouse -> Additional Layout Options -> Alt win keys -> Control is mapped to win key (and usual ctrl keys)
# Debian: Typing -> Alt win keys -> Control is mapped to win key (and usual ctrl keys)

# Settings -> Keyboard -> 
# * Switch applications is Ctrl+Tab
# * Change show all applications -> LaunchB
# * Show the activities overview - launchA
# * Lock screen - Ctrl+L

# * Change in gnome terminal/terminator copy paste close
